### Suggestions

<ComponentPreview details={<>Combine <code>TagInput</code> with clickable buttons to let users add tags by clicking suggestions or typing manually. Suggestions disappear once selected.</>} code={`const SUGGESTIONS = ['React', 'TypeScript', 'Next.js', 'Tailwind CSS', 'Node.js', 'GraphQL', 'PostgreSQL', 'Redis', 'Docker', 'AWS'];

const [tags, setTags] = useState([
  { value: 'React', valid: true },
  { value: 'TypeScript', valid: true },
]);

const selectedValues = new Set(tags.map((t) => t.value));
const available = SUGGESTIONS.filter((s) => !selectedValues.has(s));

<div className="flex flex-col gap-3">
  <TagInput value={tags} onChange={setTags} placeholder="Add a skill..." />
  {available.length > 0 && (
    <div className="flex flex-wrap gap-1.5">
      {available.map((s) => (
        <Button key={s} size="xs" color="secondary" variant="soft" pill={false}
          onClick={() => setTags([...tags, { value: s, valid: true }])}>
          {s}
        </Button>
      ))}
    </div>
  )}
</div>`} previewMinHeight={168}>
  <TagInputSuggestionsDemo />
</ComponentPreview>
