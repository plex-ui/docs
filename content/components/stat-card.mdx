## Examples

### Overview

<ComponentPreview details={<>Dashboard-ready stat cards with trends, icons, and progress bars. Use the controls below to explore all available props.</>} code={`const [size, setSize] = useState("default")
const [variant, setVariant] = useState("default")
const [trendPosition, setTrendPosition] = useState("value")
const [trendVariant, setTrendVariant] = useState("text")
const [showTrend, setShowTrend] = useState(true)
const [showTooltip, setShowTooltip] = useState(false)
const [invertTrend, setInvertTrend] = useState(false)

<StatCard
  label="Total Revenue"
  value="$45,231"
  size={size}
  variant={variant}
  trendPosition={trendPosition}
  trendVariant={trendVariant}
  trend={showTrend ? { value: 12.5, label: "vs last month" } : undefined}
  tooltip={showTooltip ? "Sum of all revenue streams" : undefined}
  invertTrend={invertTrend}
/>`} previewMinHeight={120}>
  <StatCardOverviewDemo />
</ComponentPreview>

### Sizes

  <div className="grid grid-cols-3 gap-4">

    <StatCard label="Active Users" value="2,847" trend={{ value: 8.2 }} />

  {/* Small size */}

</ComponentPreview>

### With icon

<ComponentPreview details={<>Pass an <code>icon</code> to display it on the left side with a soft background. The layout automatically switches to a horizontal icon + content arrangement. Works with both default and <code>sm</code> sizes.</>} code={`import { DollarCircle, Users, ShoppingBag } from "@plexui/ui/components/Icon"

{/* Default size */}
<div className="grid grid-cols-2 gap-4">
  <StatCard icon={<DollarCircle />} label="Total Revenue" value="$45,231" trend={{ value: 12.5 }} />
  <StatCard icon={<Users />} label="Active Users" value="2,847" trend={{ value: 8.2 }} />
</div>

{/* Small size */}
<div className="grid grid-cols-3 gap-3">
  <StatCard size="sm" icon={<DollarCircle />} label="Revenue" value="$45,231" trend={{ value: 12.5 }} />
  <StatCard size="sm" icon={<Users />} label="Users" value="2,847" trend={{ value: 8.2 }} />
  <StatCard size="sm" icon={<ShoppingBag />} label="Orders" value="1,234" trend={{ value: -3.1 }} />
</div>`} previewMinHeight={120}>
  <StatCardIconDemo />
</ComponentPreview>
