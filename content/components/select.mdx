### Column configuration

<ComponentPreview details={<>A column-visibility panel built with <code>Popover</code> and <code>Switch</code>. Toggling a switch moves the column between shown and hidden groups.</>} code={`const [shown, setShown] = useState(['name', 'email', 'role', 'status'])
const toggle = (id) => {
  setShown((prev) =>
    prev.includes(id) ? prev.filter((c) => c !== id) : [...prev, id]
  )
}
const shownColumns = allColumns.filter((c) => shown.includes(c.id))
const hiddenColumns = allColumns.filter((c) => !shown.includes(c.id))
  <Popover.Trigger>
    <Button variant="outline" size="sm"><SettingsSlider /> Columns</Button>
  </Popover.Trigger>
  <Popover.Content minWidth={240} side="bottom" align="start">
    {shownColumns.map((col) => (
      <Row key={col.id} onClick={() => toggle(col.id)}>
        <span>{col.label}</span>
        <Switch checked />
      </Row>
    ))}
    {hiddenColumns.map((col) => (
      <Row key={col.id} onClick={() => toggle(col.id)}>
        <span className="text-tertiary">{col.label}</span>
        <Switch checked={false} />
      </Row>
    ))}
  </Popover.Content>
</Popover>`} previewMinHeight={168}>
  <SelectColumnConfigDemo />
</ComponentPreview>
