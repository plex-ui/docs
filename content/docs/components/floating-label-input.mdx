### Country select

<ComponentPreview details={<>Use <code>FloatingLabelSelect</code> alongside <code>FloatingLabelInput</code> for consistent form styling. The select trigger shares the same floating-label visual shell — the label floats up when a value is selected.</>} code={`import { FloatingLabelSelect } from "@plexui/ui/components/FloatingLabelSelect"
import { Menu } from "@plexui/ui/components/Menu"

const [country, setCountry] = useState(null)
const selected = COUNTRIES.find((c) => c.value === country)

<Menu>
  <Menu.Trigger>
    <FloatingLabelSelect
      label="Country"
      selected={!!selected}
      onClearClick={() => setCountry(null)}
    >
      {selected?.flag} {selected?.label}
    </FloatingLabelSelect>
  </Menu.Trigger>
  <Menu.Content>
    <Menu.RadioGroup value={country} onChange={setCountry}>
      {COUNTRIES.map((c) => (
        <Menu.RadioItem key={c.value} value={c.value}>
          {c.flag} {c.label}
        </Menu.RadioItem>
      ))}
    </Menu.RadioGroup>
  </Menu.Content>
</Menu>`} previewMinHeight={168}>
  <FloatingLabelSelectCountryDemo />
</ComponentPreview>
