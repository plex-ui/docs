### Group with select all

<ComponentPreview details={<>Combine a parent checkbox with child checkboxes to create a "select all" pattern. The parent shows an <code>indeterminate</code> state when only some children are checked.</>} code={`const [items, setItems] = useState([
  { id: "read", label: "Read", checked: true },
  { id: "write", label: "Write", checked: false },
  { id: "execute", label: "Execute", checked: false },
]);

const allChecked = items.every(i => i.checked);
const noneChecked = items.every(i => !i.checked);
const parentChecked = allChecked ? true : noneChecked ? false : "indeterminate";

<>
  <Checkbox
    label="Permissions"
    checked={parentChecked}
    onCheckedChange={() => {
      const next = !allChecked;
      setItems(prev => prev.map(i => ({ ...i, checked: next })));
    }}
  />
  <div className="pl-6 flex flex-col gap-3">
    {items.map(item => (
      <Checkbox
        key={item.id}
        label={item.label}
        checked={item.checked}
        onCheckedChange={next => {
          setItems(prev =>
            prev.map(i => i.id === item.id ? { ...i, checked: Boolean(next) } : i)
          );
        }}
      />
    ))}
  </div>
</>`} previewMinHeight={240}>
  <CheckboxGroupDemo />
</ComponentPreview>
