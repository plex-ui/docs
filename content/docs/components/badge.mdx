### Colors & variants

<ComponentPreview details={<>You can stylize the display of the badge using the <code>color</code> and <code>variant</code> props.</>} code={`const variants = ["soft", "solid", "outline"] as const
const colors = ["secondary", "success", "warning", "danger", "info", "discovery", "caution"] as const

<div className="pt-1 pb-6 px-8 w-fit min-w-full">
  <div
    className="grid gap-6 items-center justify-center"
    style={{ gridTemplateColumns: \`auto repeat(\${colors.length}, min-content)\` }}
  >
    <div />
    {colors.map((color) => (
      <div key={color} className="text-center mb-1 text-sm" style={{ color: "var(--color-text-tertiary)" }}>
        {color}
      </div>
    ))}

    {variants.map((variant) => (
      <React.Fragment key={variant}>
        <div className="text-right mr-3 text-sm" style={{ color: "var(--color-text-tertiary)" }}>
          {variant}
        </div>
        {colors.map((color) => (
          <div key={\`\${variant}-\${color}\`} className="text-center">
            <Badge color={color} variant={variant}>Sample</Badge>
          </div>
        ))}
      </React.Fragment>
    ))}
  </div>
</div>`} previewMinHeight={168}>
  <BadgeColorsDemo />
</ComponentPreview>

### Size & roundness

<ComponentPreview details={<>You can control the size and roundness of the badge with <code>size</code> and <code>pill</code> props. When <code>pill</code> is set, extra horizontal gutter is dynamically applied to the badge.</>} code={`const [size, setSize] = useState("md")
const [pill, setPill] = useState(false)

<>
  <div style={{ background: "var(--color-surface-secondary)", width: "100%" }}>
    <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", padding: "6px 16px 6px 8px", borderTop: "1px solid var(--color-fd-border)" }}>
      <span style={{ fontFamily: "var(--font-mono, ui-monospace, monospace)", fontSize: "0.8125rem", padding: "2px 8px", borderRadius: 6, background: "var(--color-surface-secondary)" }}>
        size
      </span>
      <SegmentedControl value={size} onChange={setSize} size="xs" aria-label="size">
        <SegmentedControl.Option value="sm">sm</SegmentedControl.Option>
        <SegmentedControl.Option value="md">md</SegmentedControl.Option>
        <SegmentedControl.Option value="lg">lg</SegmentedControl.Option>
      </SegmentedControl>
    </div>
    <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", padding: "6px 16px 6px 8px", borderTop: "1px solid var(--color-fd-border)" }}>
      <span style={{ fontFamily: "var(--font-mono, ui-monospace, monospace)", fontSize: "0.8125rem", padding: "2px 8px", borderRadius: 6, background: "var(--color-surface-secondary)" }}>
        pill
      </span>
      <Switch checked={pill} onCheckedChange={setPill} aria-label="pill" />
    </div>
  </div>
  <div className="py-10 flex justify-center">
    <Badge color="info" size={size} pill={pill}>New</Badge>
  </div>
</>`} previewMinHeight={120}>
  <BadgeSizingDemo />
</ComponentPreview>
